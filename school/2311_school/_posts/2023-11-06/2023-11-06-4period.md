### 문제

'*'기호를 사용하여 삼각형 구성하기

	   *
	  ***
	 *****
	*******
```java
//제일 아랫단이 제일 먼저 출력됨 
//ㄴ 아니었음 제일 아랫단이 제일 마지막에 출력됨
//첫번째 출력에는 스페이스 3개, 별이 1개
//두번째 출력에서는 스페이스 2개 별 3개
//세번째 출력에서는 스페이스 1번 별 5개
//for문 조건( a ; b ; c )
//a: 별 변수=1, 스페이스 변수=3;
//b: 스페이스 변수가 0이 되면 끝
//c: 별 +2, 스페이스 -1
        
for(int i=1, j=3; j>=0; i+=2,j--) {
    //스페이스 출력
    for(int n = 0;n<j;n++) {
        System.out.print(" ");
    }
    //별 출력
    for(int m = 0;m<i;m++) {
        System.out.print("*");
    }
    //줄 바꾸기
    System.out.println();
} 
```

//다이아몬드 버전
int a = 2;
int b = -1;

//for 루프 안에 a,b변수를 넣어도 됨
//int i=1, j=10,a=2,b=-1;
for(int i=1, j=3; i>=1; i+=a,j+=b) {
    if(j==0) {
        a=-2;
        b=1;
    }
    for(int n = 0;n<j;n++) {
        System.out.print(" ");
    }
    for(int m = 0;m<i;m++) {
        System.out.print("*");
    }
    System.out.println();
}

//선생님 해답
//i는 줄바꿈 수 이면서 스페이스 변수에서 빼주는 수
//star는 별 변수

int space = 10;
for(int i=0, star=1;i<=3;i++, star+=2) {
    for(int j=0;j<space-i;j++) {
        System.out.print(" ");
    }
    for(int k=0;k<star;k++) {
        System.out.printf("%c",'*');
    }
    System.out.println();
}
