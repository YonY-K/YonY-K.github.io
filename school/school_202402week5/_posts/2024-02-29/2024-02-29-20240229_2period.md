### 버튼을 클릭하면 리스트가 나오는 방식

#### List.vue

```html
<template>
    <button @click="getList">리스트 보기</button>

    <table id="table">
            <tr>
                <th>사번</th>
                <th>이름</th>
                <th>직급</th>
                <th>상사</th>
                <th>입사일</th>
                <th>급여</th>
                <th>부서번호</th>
            </tr>
            <tr v-for="i in list " :key="i">
                <td>{{i.empno}}</td>
                <td>{{i.ename}}</td>
                <td>{{i.job}}</td>
                <td>{{i.mgr}}</td>
                <td>{{i.hiredate}}</td>
                <td>{{i.sal}}</td>
                <td>{{i.deptno}}</td>
            </tr>
    </table>
</template>

<script>
import axios from 'axios';

export default {
    data() {
        return {
            list : []
        }
    },
    methods : {
        async getList() {
            try{
                const response = await axios.post('http://localhost/vue/list');
                this.list = response.data;
                document.getElementById('table').style.display = 'block';
            }catch(error){
                alert('요청시 오류발생');
            }
        }
    }
}
</script>

<style>
    table {
        display: none;
    }
</style>
```

#### VueController.java

```java
package com.ezenac.vue.crud;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.CrossOrigin;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
@RequestMapping("/vue")
public class VueController {
	
	@Autowired
	private EmpMapper em;
	
	@GetMapping("")
	public String index() {
		return "VueController running...";
	}
	
	@CrossOrigin("*")
	@PostMapping("/member")
	public String getJSON(@RequestBody String strJSON) {
		System.out.println("오느냐");
		System.out.println(strJSON);
		return strJSON;
	}
	
	
	@CrossOrigin("*")
	@PostMapping("/list")
	public ResponseEntity<List<Emp>> getList() {
		List<Emp> list = em.getList();
		return new ResponseEntity<List<Emp>>(list, HttpStatus.OK);
	}
}
```

#### Emp.java

```java
package com.ezenac.vue.crud;

import java.sql.Date;

import org.springframework.stereotype.Component;

import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;

@Component
@Data
@NoArgsConstructor
@AllArgsConstructor
public class Emp {
	private int empno;
	private String ename;
	private String job;
	private String mgr;
	private Date hiredate;
	private int sal;
	private int deptno;
}
```

#### EmpMapper.java

```java
package com.ezenac.vue.crud;

import java.util.List;

import org.apache.ibatis.annotations.Mapper;
import org.apache.ibatis.annotations.Select;

@Mapper
public interface EmpMapper {

	@Select("SELECT * FROM emp")
	List<Emp> getList();
}
```
